Bekerja dengan Library Eksternal

Bagian 1: Pengenalan Library Eksternal

Apa itu Library/Package?

  • Library (atau package) adalah kumpulan kode yang sudah dibuat sebelumnya yang dapat kita gunakan untuk memperluas kemampuan Python.
  • Analogi: Library seperti perkakas yang siap pakai
  • Python standar -> Perlengkapan dasar
  • Library eksternal -> Perkakas khusus untuk tugas tertentu

Mengapa Menggunakan Library Eksternal?

Daripada menulis semuanya dari awal, kita bisa:

  • Menghemat waktu
  • Mengurangi bug
  • Memanfaatkan kode yang sudah teruji
  • Fokus pada logika bisnis

Bagian 2: Package Manager - pip

Apa itu pip?

  • pip adalah package manager untuk Python yang digunakan untuk menginstall dan mengelola library eksternal.

    name TEXT NOT NULL,
    department TEXT,
    salary REAL,
    hire_date TEXT
)
''')

print("\nNested Progress Bars:")

    for j in tqdm(range(10), desc=f"Inner {i}", leave=False):

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime
import requests

class COVIDAnalyzer:
    def __init__(self):
        self.df = None
    
    def load_data(self):
        """Load data COVID-19 dari API"""
        try:
            # URL contoh (ganti dengan API yang aktual)
            url = "https://api.covid19api.com/summary"
            response = requests.get(url)
            data = response.json()
            
            # Convert ke DataFrame
            self.df = pd.DataFrame(data['Countries'])
            print(f"Data loaded: {len(self.df)} countries")
            
        except Exception as e:
            print(f"Error loading data: {e}")
            # Fallback ke data sample
            self._create_sample_data()
    
    def _create_sample_data(self):
        """Membuat sample data untuk demo"""
        data = {
            'Country': ['Indonesia', 'Malaysia', 'Singapore', 'Thailand', 'Vietnam'],
            'TotalConfirmed': [6000000, 3000000, 1500000, 2500000, 3500000],
            'TotalDeaths': [150000, 35000, 1200, 22000, 41000],
            'TotalRecovered': [5800000, 2950000, 1490000, 2470000, 3450000],
            'Date': ['2024-01-01'] * 5
        }
        self.df = pd.DataFrame(data)
        print("Using sample data")
    
    def analyze_data(self):
        """Analisis data COVID-19"""
        if self.df is None:
            print("Data belum diload!")
            return
        
        # Basic statistics
        print("=== STATISTIK DASAR ===")
        print(f"Total negara: {len(self.df)}")
        print(f"Total confirmed cases: {self.df['TotalConfirmed'].sum():,}")
        print(f"Total deaths: {self.df['TotalDeaths'].sum():,}")
        
        # Top 10 countries by cases
        top_10 = self.df.nlargest(10, 'TotalConfirmed')[['Country', 'TotalConfirmed', 'TotalDeaths']]
        print("\n=== 10 NEGARA DENGAN KASUS TERBANYAK ===")
        print(top_10.to_string(index=False))
        
        # Calculate death rate
        self.df['DeathRate'] = (self.df['TotalDeaths'] / self.df['TotalConfirmed'] * 100).round(2)
        
        # Countries with highest death rate (min 1000 cases)
        high_death_rate = self.df[self.df['TotalConfirmed'] > 1000].nlargest(5, 'DeathRate')
        print("\n=== 5 NEGARA DENGAN DEATH RATE TERTINGGI ===")
        print(high_death_rate[['Country', 'DeathRate', 'TotalConfirmed']].to_string(index=False))
    
    def visualize_data(self):
        """Visualisasi data COVID-19"""
        if self.df is None:
            print("Data belum diload!")
            return
        
        plt.figure(figsize=(15, 10))
        
        # Plot 1: Top 10 countries by cases
        plt.subplot(2, 2, 1)
        top_10 = self.df.nlargest(10, 'TotalConfirmed')

        plt.title('10 Negara dengan Kasus Terbanyak')
        plt.xlabel('Total Kasus')
        
        # Plot 2: Death rate distribution
        plt.subplot(2, 2, 2)

        plt.xlabel('Death Rate (%)')

        plt.title('Korelasi Variabel COVID-19')
        
        # Plot 4: Cases vs Deaths scatter
        plt.subplot(2, 2, 4)

        plt.xlabel('Total Kasus')
        plt.ylabel('Total Kematian')
        plt.title('Kasus vs Kematian')
        plt.xscale('log')
        plt.yscale('log')
        
        plt.tight_layout()

import requests
from requests.exceptions import RequestException
import pandas as pd
import numpy as np

def safe_api_call(url):
    """Wrapper aman untuk API calls"""
    try:
        response = requests.get(url, timeout=10)
        response.raise_for_status()  # Raise exception untuk HTTP errors
        return response.json()
    except RequestException as e:
        print(f"Error accessing {url}: {e}")
        return None

def safe_data_processing(df):
    """Wrapper aman untuk data processing"""
    try:
        # Contoh operasi yang mungkin error
        result = df.groupby('category')['value'].mean()
        return result
    except (KeyError, ValueError, TypeError) as e:
        print(f"Error processing data: {e}")
        return pd.Series()  # Return empty series sebagai fallback

# Contoh penggunaan
data = safe_api_call('https://api.example.com/data')
if data:
    df = pd.DataFrame(data)
    result = safe_data_processing(df)
    print(result)
